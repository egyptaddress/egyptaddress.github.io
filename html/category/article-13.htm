安装gcc

1，准备安装包

下载地址 http://nginx.org/download/nginx-1.19.0.tar.gz

rmp 下载地址 http://mirror.centos.org/centos/7/os/x86_64/Packages/

rpm -Uvh pcre-8.32-17.el7.x86_64.rpm
rpm -Uvh pcre-devel-8.32-17.el7.x86_64.rpm
rpm -Uvh zlib-1.2.7-18.el7.x86_64.rpm
rpm -Uvh zlib-devel-1.2.7-18.el7.x86_64.rpm
rpm -Uvh openssl-1.0.2k-19.el7.x86_64.rpm
rpm -Uvh keyutils-libs-devel-1.5.8-3.el7.x86_64.rpm
rpm -Uvh --nodeps libcom_err-1.42.9-17.el7.x86_64.rpm
rpm -Uvh libcom_err-devel-1.42.9-17.el7.x86_64.rpm
rpm -Uvh krb5-libs-1.15.1-46.el7.x86_64.rpm
rpm -Uvh libkadm5-1.15.1-46.el7.x86_64.rpm
rpm -Uvh --nodeps krb5-devel-1.15.1-37.el7_6.x86_64.rpm
rpm -Uvh openssl-devel-1.0.2k-19.el7.x86_64.rpm

2，安装依赖
不考虑依赖关系安装所有，你也可以一个个安装

rpm -Uvh --nodeps *.rpm

3，安装nignx

tar -zxvf nginx-1.19.0.tar.gz
cd nginx-1.19.0
#1
./configure --prefix=/usr/local/nginx --user=nginx --group=nginx --with-http_stub_status_module --with-http_ssl_module --with-http_sub_module --with-http_gzip_static_module --with-stream
#2.
./configure --prefix=/usr/local/nginx --with-http_ssl_module --with-http_realip_module --with-http_addition_module --with-http_sub_module --with-http_dav_module --with-http_flv_module --with-http_mp4_module --with-http_gunzip_module --with-http_gzip_static_module --with-http_random_index_module --with-http_secure_link_module --with-http_stub_status_module --with-http_auth_request_module --with-threads --with-stream --with-stream_ssl_module --with-http_slice_module --with-file-aio --with-http_v2_module --with-ipv6
make && make install
cd /usr/local/nginx/sbin
./nginx -v
