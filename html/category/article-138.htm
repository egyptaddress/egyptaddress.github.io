import pandas as pd df = pd.DataFrame() for chunk in pd.read_csv('test.csv', usecold=[0,1], names=['times', 'data'], chunksize=2000000):     df = df.append(chunk)

chunksize 尽量不要超过1000w数据，超过1000w数据响应时间比200w拼接时间长好几倍
